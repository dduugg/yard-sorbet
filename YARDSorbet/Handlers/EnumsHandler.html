def process
  statement.traverse do |node|
    if const_assign_node?(node)
      register YARD::CodeObjects::ConstantObject.new(namespace, node.first.source) do |obj|
        obj.docstring = node.docstring
        obj.source = node
        obj.value = node.last.source
      end
    end
  end
end
