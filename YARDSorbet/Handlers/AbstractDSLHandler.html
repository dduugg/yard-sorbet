def process
  return if namespace.has_tag?(:abstract)

  text = namespace.is_a?(YARD::CodeObjects::ClassObject) ? CLASS_TAG_TEXT : TAG_TEXT
  tag = YARD::Tags::Tag.new(:abstract, text)
  namespace.add_tag(tag)
end
